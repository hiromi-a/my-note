grid-template-columns: 1fr 1fr 1fr 1fr;
grid-gap: 0 28px;

    grid-template-columns: repeat(4, 1fr);

    grid-template-rows: repeat(3, 1fr);
  • 同じ幅4つ
    grid-template-columns: 1fr 1fr 1fr 1fr;
  • 同じ高さ3つ
    grid-template-rows: 1fr 1fr 1fr;
